UINT AFXAPI HashKey(LPCWSTR key)
{
	ENSURE_ARG(AfxIsValidString(key));
	UINT nHash = 0;
	while (*key)
		nHash = (nHash<<5) + nHash + *key++;
	return nHash;
}
